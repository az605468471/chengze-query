# 基于区块链的在线教学平台方案

## 🎯 项目概述

### 核心理念
**去中心化、透明、可信**的在线教育平台，利用区块链技术解决传统教育平台的信任问题。

### 差异化优势
| 传统平台 | 区块链平台 |
|----------|------------|
| 证书易伪造 | 链上证书，不可篡改 |
| 收入分配不透明 | 智能合约自动分账 |
| 用户数据被平台控制 | 用户拥有数据主权 |
| 课程质量难验证 | 链上评价，真实可信 |

## 🔧 核心功能模块

### 1. 课程管理模块
```solidity
// 课程注册合约
contract CourseRegistry {
    struct Course {
        address instructor;      // 讲师地址
        string  title;           // 课程标题
        string  description;     // 课程描述
        uint256 price;           // 价格 (wei)
        string  ipfsHash;        // 课程资源IPFS地址
        bool    isActive;        // 是否上架
    }
    
    mapping(uint256 => Course) public courses;
    uint256 public courseCount;
    
    // 创建课程
    function createCourse(
        string memory _title,
        string memory _description,
        uint256 _price,
        string memory _ipfsHash
    ) public returns (uint256) {
        courseCount++;
        courses[courseCount] = Course({
            instructor: msg.sender,
            title: _title,
            description: _description,
            price: _price,
            ipfsHash: _ipfsHash,
            isActive: true
        });
        return courseCount;
    }
}
```

### 2. 支付分配模块
```solidity
// 支付分配合约
contract PaymentDistribution {
    // 讲师获得 70%，平台 30%
    uint256 public constant INSTRUCTOR_SHARE = 70;
    uint256 public constant PLATFORM_SHARE = 30;
    
    mapping(uint256 => uint256) public courseEarnings;
    mapping(address => uint256) public instructorEarnings;
    
    // 购买课程
    function purchaseCourse(uint256 _courseId) public payable {
        Course storage course = courses[_courseId];
        require(msg.value >= course.price, "Insufficient payment");
        
        // 计算分配
        uint256 instructorAmount = (msg.value * INSTRUCTOR_SHARE) / 100;
        uint256 platformAmount = (msg.value * PLATFORM_SHARE) / 100;
        
        // 转账
        payable(course.instructor).transfer(instructorAmount);
        payable(platformWallet).transfer(platformAmount);
        
        // 记录
        courseEarnings[_courseId] += msg.value;
        instructorEarnings[course.instructor] += instructorAmount;
    }
}
```

### 3. 证书颁发模块
```solidity
// 证书合约 (ERC-721)
contract CourseCertificate is ERC721 {
    struct Certificate {
        uint256 courseId;
        address student;
        uint256 completionDate;
        string  metadataURI;  // 证书详情IPFS地址
    }
    
    mapping(uint256 => Certificate) public certificates;
    uint256 public certificateCount;
    
    // 颁发证书
    function issueCertificate(
        uint256 _courseId,
        address _student,
        string memory _metadataURI
    ) public onlyInstructor(_courseId) returns (uint256) {
        certificateCount++;
        _mint(_student, certificateCount);
        certificates[certificateCount] = Certificate({
            courseId: _courseId,
            student: _student,
            completionDate: block.timestamp,
            metadataURI: _metadataURI
        });
        return certificateCount;
    }
}
```

### 4. 评价系统模块
```solidity
// 评价合约
contract ReviewSystem {
    struct Review {
        address student;
        uint256 courseId;
        uint8   rating;        // 1-5分
        string  comment;
        uint256 timestamp;
    }
    
    mapping(uint256 => Review[]) public courseReviews;
    mapping(uint256 => uint256) public courseAverageRating;
    
    // 提交评价
    function submitReview(
        uint256 _courseId,
        uint8 _rating,
        string memory _comment
    ) public {
        require(_rating >= 1 && _rating <= 5, "Invalid rating");
        require(hasPurchasedCourse(_courseId, msg.sender), "Not purchased");
        
        courseReviews[_courseId].push(Review({
            student: msg.sender,
            courseId: _courseId,
            rating: _rating,
            comment: _comment,
            timestamp: block.timestamp
        }));
        
        // 更新平均分
        updateAverageRating(_courseId);
    }
}
```

### 1. 技术风险
- **Gas费用**：BSC Gas较低，但仍有成本
- **用户体验**：钱包连接复杂
- **扩展性**：区块链性能限制

### 2. 市场风险
- **用户教育**：需要教育用户使用钱包
- **竞争激烈**：已有多个区块链教育平台
- **监管政策**：加密货币监管不确定性

### 3. 运营风险
- **内容质量**：需要建立审核机制
- **讲师管理**：需要建立激励机制
- **用户留存**：需要建立社区
